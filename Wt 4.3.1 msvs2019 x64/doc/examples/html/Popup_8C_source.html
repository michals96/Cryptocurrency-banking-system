29  *

40  + cancelPressed_.createCall({}) +
41  "}}");
42  break;
43  case Alert:
44  show.setJavaScript

46  + okPressed_.createCall({"''"}) +
47  "}");
48  break;
49  case Prompt:
50  show.setJavaScript

52  + defaultValue_ + "');"
